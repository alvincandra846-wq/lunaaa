  <script>
    const heart = document.getElementById('heart');
    const tap = document.getElementById('tap');
    const message = document.getElementById('message');
    const audio = document.getElementById('bgm');

    window.onload = () => {
      heart.style.opacity = 1;
      tap.style.opacity = 1;
    }

    document.body.addEventListener('click', (e) => {
      // Buat efek lopyu di lokasi tap
      const lopyu = document.createElement('div');
      lopyu.className = 'lopyu';
      lopyu.style.left = `${e.clientX - 30}px`;
      lopyu.style.top = `${e.clientY - 20}px`;
      lopyu.textContent = 'lopyuuuuuu';
      document.body.appendChild(lopyu);

      setTimeout(() => {
        lopyu.remove();
      }, 1000);

      // Sembunyikan elemen awal & tampilkan pesan
      heart.style.opacity = 0;
      tap.style.display = 'none';

      setTimeout(() => {
        message.style.opacity = 1;
        audio.play();
        startHeartRain();
      }, 600);
    });

    function startHeartRain() {
      const symbols = ['üíó', 'üíû', 'üíê üå∏ üíû'];
      const interval = setInterval(() => {
        const heart = document.createElement('div');
        heart.className = 'heart-fall';
        heart.textContent = symbols[Math.floor(Math.random() * symbols.length)];
        heart.style.left = Math.random() * 100 + 'vw';
        heart.style.animationDuration = (2 + Math.random() * 3) + 's';
        document.body.appendChild(heart);
        setTimeout(() => {
          heart.remove();
        }, 5000);
      }, 200);

      // Stop after 10 seconds
      setTimeout(() => clearInterval(interval), 10000);
    }
